Define Measure '銷售明細'[去重複客戶數] = DistinctCount('客戶別'[CustomerId])

Evaluate
 Summarize (
	'銷售明細'
		  , '銷售日別'[年度]
		  , '產品別'[產品大類]
		  , "總銷售金額", Format(Sum('銷售明細'[銷售金額]),"#,###")
		  , "去重複客戶數", Format([去重複客戶數],"#,###")
		  , "人均年度消費",Format([總銷售金額]/[去重複客戶數],"#,###.#")
		   )
 Order by '銷售日別'[年度],'產品別'[產品大類]
